# Bind mounts from host into container (project-wide, committed).
# Format: <host-path>:<container-path>[:<options>]
# ~ at the start of host-path expands to $HOME.
# Missing host paths are skipped with a warning.
# See list-file.sh for comment and blank-line handling.
#
# Per-developer mounts go in host-mounts.local.txt (gitignored).
# Example:
#   ~/.config/xorq:/home/vscode/.config/xorq:ro
